defunc (tester [expr]
	echo_rich("[color=green][b]================================[/b][/color]")
	echo_raw(expr)
	echo_rich("[color=teal]↑↑ Raw ↑↑ --------- ↓↓ Evaled ↓↓[/color]")
	echo_val(eval(expr))
)

tester (raw(
	func ([] +(1 2))
))

tester (raw(
	func ([x y]
		#((1 * 6 + x) * y))
))

defvar (:const *const_var* "hello")

defunc (:const const_fn [] " world")

defunc (:const :pure pure_fn [name] #(*const_var* + " " + name))

tester (raw(
	func ([x]
		#(x + (*const_var* + #:const_fn()) + #:pure_fn("gss")))
))

tester (raw(
	func ([x y]
		#(x #:factor([x y] +(x y)) y))
))
