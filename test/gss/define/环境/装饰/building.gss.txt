defvar(building/make func([ref x y id]

		id id
		props {
			need_route 	#f
			routable 	#f
			collision_boxes [rect2(-1.5 -1.5 3 2)]
			asserts {

				building [&static &main clip3x5_build(x y)]
			}
			groups prop/pushs([2409])
		}
	)
))
