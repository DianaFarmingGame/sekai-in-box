(defvar 食物 (make_define GItem
	prop {
		groups (prop/pushs [&物品])
		name "食物"
		}
			)
)

#面包
(define 食物
	ref 1000
	id &物品/自用道具/面包 ; 
	prop {
		name "面包"
		description "新鲜松软的白面包"
		groups (prop/pushs ["食物"])
		item_icon [&static &main (clip_item 0 0)]
		pickable #t ; 能否取得
		stackable #t ; 能否堆叠
		stack_capacity 64 ; 堆叠上限
		stack_count 1 ; 堆叠数量
		lp 1 ; 可用度
		chargeable #f ; 能否修复
		tradeable #t ; 能否交易
		price 24 ; 价格
		action {
			default & ; 攻击
			smash & ; 破坏方块
				}
		}
)
