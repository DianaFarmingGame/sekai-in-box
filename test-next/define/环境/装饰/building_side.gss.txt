defvar (building_side/make func([ref x y id]
	define/make (load ("entity_block.gd")
		ref ref
		id id
		props {
			need_route 	#f
			routable 	#f
			collision_boxes [rect2(-1.5 -1.5 3 2)]
			asserts {
				main load ("../房子.png")
			}
			cur_draw &building_side
			draw_data {
				building_side [&static &main clip3x3_build(x y)]
			}
		}
		groups prop/pushs([2415])
	)
))
